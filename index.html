  1. 打開您的 Google 雲端硬碟，新增一個「Google 試算表 > 空白試算表」。

  5. 清空原有程式碼，貼上黑框代碼並點擊「儲存」。 
    function doGet(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  
  // 支援網頁端隨時單獨檢查「最後更新時間」
  if (e.parameter.action === "checkVersion") {
    var lastUpdateTime = sheet.getRange("A2").getValue();
    return ContentService.createTextOutput(JSON.stringify({ lastUpdateTime: lastUpdateTime })).setMimeType(ContentService.MimeType.JSON);
  }
  
  // 原本的讀取邏輯 (讀取 A1) ➔ 升級：連同 A2 時間一起回傳
  var data = sheet.getRange("A1").getValue(); 
  var lastUpdateTime = sheet.getRange("A2").getValue();
  var result = {};
  if (data) { 
    result = { 
      JSONData: data,
      lastUpdateTime: lastUpdateTime // 讓網頁一登入就知道雲端的最新時間
    }; 
  }
  return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  
  // 直接解析傳過來的純文字 JSON
  var postData = JSON.parse(e.postData.contents);
  
  // 網頁儲存前，檢查 A2 儲存格的時間
  if (postData.action === "checkVersionBeforeSave") {
    var cloudTime = new Date(sheet.getRange("A2").getValue()).getTime();
    var clientTime = new Date(postData.clientLoadedTime).getTime();
    
    // 如果雲端時間大於網頁上次同步嘅時間，代表網頁資料過期
    if (cloudTime > clientTime) {
      var data = sheet.getRange("A1").getValue();
      return ContentService.createTextOutput(JSON.stringify({ 
        status: "outdated", 
        message: "發現雲端數據已被更新！",
        JSONData: data,
        lastUpdateTime: sheet.getRange("A2").getValue()
      })).setMimeType(ContentService.MimeType.JSON);
    }
  }

  // 安全或直接儲存邏輯 (寫入 A1)
  sheet.getRange("A1").setValue(postData.JSONData); 
  
  // 每次網頁更新儲存時，將當前時間寫入 A2 儲存格
  sheet.getRange("A2").setValue(new Date().toISOString());
  
  // 自動觸發重新排版
  parseJsonToTable();
  
  return ContentService.createTextOutput(JSON.stringify({ status: "success" })).setMimeType(ContentService.MimeType.JSON);
}

// 建立 Google Sheet 自訂選單
function onOpen() {
  var ui = SpreadsheetApp.getUi();
  ui.createMenu('📊 網購數據管理')
    .addItem('手動更新：JSON ➔ 表格', 'parseJsonToTable')
    .addItem('手動更新：表格 ➔ JSON', 'parseTableToJson')
    .addToUi();
}

// 【監聽器】人手改 Google Sheet 欄位時觸發
function onEdit(e) {
  var range = e.range;
  var row = range.getRow();
  var col = range.getColumn();
  
  // 確保修改位置在資料區（第 4 行以下，B 欄到 R 欄之間）
  if (row >= 4 && col >= 2 && col <= 18) {
    parseTableToJson();
  }
}

// 【同步一】讀取 A1 JSON ➔ 填入 B3 表格
function parseJsonToTable() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var rawData = sheet.getRange("A1").getValue();
  if (!rawData) return;
  
  var data = JSON.parse(rawData);
  var orders = data.orders || [];
  
  var lastRow = sheet.getLastRow();
  var lastColumn = sheet.getLastColumn();
  if (lastRow >= 3) {
    sheet.getRange(3, 2, lastRow - 2, Math.max(lastColumn - 1, 17)).clear(); 
  }
  
  var headers = ["訂單ID", "日期", "項目名稱", "預支金額", "運費", "物流方式", "總物流狀態", "發票狀態", "已交單", "已報銷", "貨品名稱", "貨品價格", "貨品貨幣", "港幣價格", "購買平台", "貨品物流", "更新日期"];
  sheet.getRange(3, 2, 1, headers.length).setValues([headers]);
  
  var rowDataList = [];
  orders.forEach(function(order) {
    var items = order.items || [];
    items.forEach(function(item) {
      rowDataList.push([
        order.id, order.date, order.name, order.advancePay, order.shippingFee,
        order.logisticsType, order.logisticsStatus, order.invoiceStatus,
        order.submitted ? "✅" : "❌", order.claim ? "✅" : "❌",
        item.name, item.price, item.currency, item.priceHKD, item.platform,
        item.itemLogisticsStatus, item.updateDate
      ]);
    });
  });
  
  if (rowDataList.length > 0) {
    sheet.getRange(4, 2, rowDataList.length, headers.length).setValues(rowDataList);
  }
  
  sheet.getRange(3, 2, 1, headers.length).setBackground("#2ecc71").setFontColor("white").setFontWeight("bold");
  for (var i = 2; i <= headers.length + 1; i++) {
    sheet.autoResizeColumn(i);
  }
}

// 【同步二】讀取 B4 表格 ➔ 重新組裝打包回 A1 JSON，並更新 A2 時間
function parseTableToJson() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var lastRow = sheet.getLastRow();
  if (lastRow < 4) return;
  
  var rawData = sheet.getRange("A1").getValue();
  var baseData = { currencies: ["人民幣", "港幣"], orders: [], password: "623" };
  if (rawData) {
    try { baseData = JSON.parse(rawData); } catch(e) {}
  }
  
  var tableValues = sheet.getRange(4, 2, lastRow - 3, 17).getValues();
  var orderMap = {};
  
  tableValues.forEach(function(row) {
    var orderId = String(row[0]).trim();
    if (!orderId) return;
    
    if (!orderMap[orderId]) {
      orderMap[orderId] = {
        id: orderId,
        date: row[1] ? (row[1] instanceof Date ? row[1].toISOString().split('T')[0] : String(row[1])) : "",
        name: String(row[2]),
        advancePay: Number(row[3]) || 0,
        shippingFee: Number(row[4]) || 0,
        logisticsType: String(row[5]),
        logisticsStatus: String(row[6]),
        invoiceStatus: String(row[7]),
        submitted: row[8] === "✅" || row[8] === true || String(row[8]).toLowerCase() === "true",
        claim: row[9] === "✅" || row[9] === true || String(row[9]).toLowerCase() === "true",
        items: []
      };
      
      var oldOrder = (baseData.orders || []).find(function(o) { return o.id === orderId; });
      if (oldOrder) {
        if (oldOrder.hasOwnProperty("claim")) orderMap[orderId]["claim"] = oldOrder["claim"];
      }
    }
    
    if (row[10]) {
      orderMap[orderId].items.push({
        name: String(row[10]),
        price: String(row[11]),
        currency: String(row[12]),
        priceHKD: String(row[13]),
        platform: String(row[14]),
        itemLogisticsStatus: String(row[15]),
        updateDate: row[16] ? (row[16] instanceof Date ? row[16].toISOString().split('T')[0] : String(row[16])) : ""
      });
    }
  });
  
  var newOrdersArray = [];
  for (var key in orderMap) {
    if (orderMap.hasOwnProperty(key)) {
      newOrdersArray.push(orderMap[key]);
    }
  }
  baseData.orders = newOrdersArray;
  
  sheet.getRange("A1").setValue(JSON.stringify(baseData));
  sheet.getRange("A2").setValue(new Date().toISOString());
}
